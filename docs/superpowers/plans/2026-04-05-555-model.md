```python
def find_swing_fractals(arrs):
    """
    Identify swing highs and swing lows using simple 3-bar fractals.
    Swing high: candle whose high > both prior and next candle highs.
    Swing low: candle whose low < both prior and next candle lows.

    Returns dict with:
      - swing_highs: list of (index, ts_ns, level)
      - swing_lows: list of (index, ts_ns, level)
    """
    highs = arrs['high']
    lows = arrs['low']
    ts_ns = arrs['ts_ns']
    n = len(highs)

    swing_highs = []
    swing_lows = []

    for i in range(1, n - 1):
        if highs[i] > highs[i - 1] and highs[i] > highs[i + 1]:
            swing_highs.append((i, int(ts_ns[i]), float(highs[i])))
        if lows[i] < lows[i - 1] and lows[i] < lows[i + 1]:
            swing_lows.append((i, int(ts_ns[i]), float(lows[i])))

    return {'swing_highs': swing_highs, 'swing_lows': swing_lows}
```

```python
def detect_smt_divergences(nq_arrs, es_arrs, nq_fractals, es_fractals):
    """
    Detect SMT divergence: NQ sweeps a 5m swing high/low but ES does NOT
    sweep its corresponding swing.

    For each NQ 5m bar, check if it sweeps a prior NQ swing:
      - Bearish SMT (SHORT): NQ high > prior swing high, but ES high does NOT
        exceed its nearest prior swing high in the same session
      - Bullish SMT (LONG): NQ low < prior swing low, but ES low does NOT
        break below its nearest prior swing low in the same session

    Returns list of dicts:
      {ts_ns, bar_idx, direction, nq_swing_level, sweep_price}
    """
    nq_ts = nq_arrs['ts_ns']
    nq_high = nq_arrs['high']
    nq_low = nq_arrs['low']
    nq_trade_date = nq_arrs['trade_date']

    es_ts = es_arrs['ts_ns']
    es_high = es_arrs['high']
    es_low = es_arrs['low']

    NS_PER_MIN = np.int64(60_000_000_000)

    # Build arrays of swing levels for fast lookup
    nq_sh = np.array([(idx, ts, lvl) for idx, ts, lvl in nq_fractals['swing_highs']],
                     dtype=[('idx', 'i4'), ('ts', 'i8'), ('lvl', 'f8')])
    nq_sl = np.array([(idx, ts, lvl) for idx, ts, lvl in nq_fractals['swing_lows']],
                     dtype=[('idx', 'i4'), ('ts', 'i8'), ('lvl', 'f8')])
    es_sh = np.array([(idx, ts, lvl) for idx, ts, lvl in es_fractals['swing_highs']],
                     dtype=[('idx', 'i4'), ('ts', 'i8'), ('lvl', 'f8')])
    es_sl = np.array([(idx, ts, lvl) for idx, ts, lvl in es_fractals['swing_lows']],
                     dtype=[('idx', 'i4'), ('ts', 'i8'), ('lvl', 'f8')])

    divergences = []

    for i in range(2, len(nq_ts)):
        curr_ts = nq_ts[i]
        curr_date = nq_trade_date[i]

        # Find most recent NQ swing high BEFORE this bar
        sh_mask = nq_sh['ts'] < curr_ts
        if sh_mask.any():
            latest_sh = nq_sh[sh_mask][-1]
            # Check if NQ sweeps this swing high
            if nq_high[i] > latest_sh['lvl']:
                # Find corresponding ES swing high (nearest in time before this bar)
                es_sh_mask = es_sh['ts'] < curr_ts
                if es_sh_mask.any():
                    es_latest_sh = es_sh[es_sh_mask][-1]
                    # Find ES bar at same timestamp
                    es_idx = int(np.searchsorted(es_ts, curr_ts, side='left'))
                    if es_idx < len(es_ts) and abs(es_ts[es_idx] - curr_ts) < 5 * NS_PER_MIN:
                        # ES did NOT sweep its swing high = bearish SMT
                        if es_high[es_idx] <= es_latest_sh['lvl']:
                            divergences.append({
                                'ts_ns': int(curr_ts),
                                'bar_idx': i,
                                'direction': 'SHORT',
                                'nq_swing_level': float(latest_sh['lvl']),
                                'sweep_price': float(nq_high[i]),
                            })

        # Find most recent NQ swing low BEFORE this bar
        sl_mask = nq_sl['ts'] < curr_ts
        if sl_mask.any():
            latest_sl = nq_sl[sl_mask][-1]
            # Check if NQ sweeps this swing low
            if nq_low[i] < latest_sl['lvl']:
                # Find corresponding ES swing low
                es_sl_mask = es_sl['ts'] < curr_ts
                if es_sl_mask.any():
                    es_latest_sl = es_sl[es_sl_mask][-1]
                    es_idx = int(np.searchsorted(es_ts, curr_ts, side='left'))
                    if es_idx < len(es_ts) and abs(es_ts[es_idx] - curr_ts) < 5 * NS_PER_MIN:
                        # ES did NOT sweep its swing low = bullish SMT
                        if es_low[es_idx] >= es_latest_sl['lvl']:
                            divergences.append({
                                'ts_ns': int(curr_ts),
                                'bar_idx': i,
                                'direction': 'LONG',
                                'nq_swing_level': float(latest_sl['lvl']),
                                'sweep_price': float(nq_low[i]),
                            })

    return divergences
```

```python
def find_ifvg(arrs, start_ts_ns, direction, min_gap_pts=MIN_IFVG_PTS):
    """
    Scan forward from start_ts_ns for a 3-candle FVG that gets inverted.

    For LONG (looking for bearish FVG that inverts into support):
      - Bearish FVG: candle1.high > candle3.low (gap down)
      - Inversion: a subsequent bar's high trades up into/through the gap
      - IFVG zone: [candle3.low, candle1.high]

    For SHORT (looking for bullish FVG that inverts into resistance):
      - Bullish FVG: candle1.low < candle3.high (gap up)
      - Inversion: a subsequent bar's low trades down into/through the gap
      - IFVG zone: [candle3.high, candle1.low]

    Returns (ifvg_ts_ns, ifvg_high, ifvg_low) or (None, None, None).
    ifvg_ts_ns = timestamp of the bar that inverted the FVG.
    """
    ts_ns = arrs['ts_ns']
    opens = arrs['open']
    highs = arrs['high']
    lows = arrs['low']
    closes = arrs['close']
    n = len(ts_ns)

    start = int(np.searchsorted(ts_ns, start_ts_ns, side='left'))
    if start >= n - 3:
        return None, None, None

    if direction == 'LONG':
        # Look for bearish FVG: candle1.high > candle3.low (gap down)
        for i in range(start, n - 2):
            c1_high = highs[i]
            c3_low = lows[i + 2]
            gap = c1_high - c3_low
            if gap >= min_gap_pts:
                # FVG found. Now scan forward for inversion (bar trades back up through gap)
                ifvg_top = float(c1_high)
                ifvg_bot = float(c3_low)
                for j in range(i + 3, n):
                    if highs[j] >= ifvg_bot:
                        # Bar entered/passed through the gap zone = inverted
                        return int(ts_ns[j]), ifvg_top, ifvg_bot
                break  # only need first FVG

    else:  # SHORT
        # Look for bullish FVG: candle1.low < candle3.high (gap up)
        for i in range(start, n - 2):
            c3_high = highs[i + 2]
            c1_low = lows[i]
            gap = c3_high - c1_low
            if gap >= min_gap_pts:
                # FVG found. Scan forward for inversion (bar trades down through gap)
                ifvg_top = float(c3_high)
                ifvg_bot = float(c1_low)
                for j in range(i + 3, n):
                    if lows[j] <= ifvg_top:
                        return int(ts_ns[j]), ifvg_top, ifvg_bot
                break

    return None, None, None
```

```python
def detect_setups(nq_1m_arrs, nq_5m_arrs, nq_3m_arrs, es_5m_arrs,
                  nq_fractals, es_fractals):
    """
    Detect all 555 setups: SMT → (IFVG + CISD in any order) → Entry.

    Returns (rows, pending):
      rows — list of dicts with all trade metadata
      pending — list of dicts for outcome resolution
    """
    NS_PER_MIN = np.int64(60_000_000_000)

    m1_ts = nq_1m_arrs['ts_ns']
    m1_high = nq_1m_arrs['high']
    m1_low = nq_1m_arrs['low']
    m1_hr = nq_1m_arrs['hr']
    m1_mn = nq_1m_arrs['mn']
    m1_dow = nq_1m_arrs['dow']
    m1_date = nq_1m_arrs['trade_date']

    # Precompute hourly range lookup: (date, hr) → high-low pts
    _hr_range = {}
    _prev_date_hr = (None, None)
    _hr_hi = _hr_lo = 0.0
    for _k in range(len(m1_ts)):
        _dh = (m1_date[_k], m1_hr[_k])
        if _dh != _prev_date_hr:
            if _prev_date_hr[0] is not None:
                _hr_range[_prev_date_hr] = _hr_hi - _hr_lo
            _prev_date_hr = _dh
            _hr_hi = m1_high[_k]
            _hr_lo = m1_low[_k]
        else:
            if m1_high[_k] > _hr_hi: _hr_hi = m1_high[_k]
            if m1_low[_k] < _hr_lo: _hr_lo = m1_low[_k]
    if _prev_date_hr[0] is not None:
        _hr_range[_prev_date_hr] = _hr_hi - _hr_lo

    # Detect SMT divergences
    divergences = detect_smt_divergences(nq_5m_arrs, es_5m_arrs, nq_fractals, es_fractals)
    print(f"   {len(divergences):,} SMT divergences detected")

    rows = []
    pending = []
    seen_dates_dir = set()  # deduplicate: one setup per date+direction

    for div in divergences:
        smt_ts = div['ts_ns']
        direction = div['direction']

        # Session filter: SMT must be in RTH
        smt_5m_idx = div['bar_idx']
        smt_hr = nq_5m_arrs['hr'][smt_5m_idx]
        smt_date = nq_5m_arrs['trade_date'][smt_5m_idx]

        # Deduplicate per date+direction (take first SMT of the day per direction)
        dedup_key = (str(smt_date), direction)
        if dedup_key in seen_dates_dir:
            continue
        seen_dates_dir.add(dedup_key)

        hr_float = float(smt_hr)
        if hr_float < SESSION_HRS[0] or hr_float >= SESSION_HRS[1]:
            continue

        # Find IFVG (check both 3m and 5m, take earliest)
        ifvg_ts_5m, ifvg_hi_5m, ifvg_lo_5m = find_ifvg(nq_5m_arrs, smt_ts, direction)
        ifvg_ts_3m, ifvg_hi_3m, ifvg_lo_3m = find_ifvg(nq_3m_arrs, smt_ts, direction)

        # Pick earliest IFVG
        ifvg_ts, ifvg_hi, ifvg_lo = None, None, None
        if ifvg_ts_5m is not None and ifvg_ts_3m is not None:
            if ifvg_ts_3m <= ifvg_ts_5m:
                ifvg_ts, ifvg_hi, ifvg_lo = ifvg_ts_3m, ifvg_hi_3m, ifvg_lo_3m
            else:
                ifvg_ts, ifvg_hi, ifvg_lo = ifvg_ts_5m, ifvg_hi_5m, ifvg_lo_5m
        elif ifvg_ts_5m is not None:
            ifvg_ts, ifvg_hi, ifvg_lo = ifvg_ts_5m, ifvg_hi_5m, ifvg_lo_5m
        elif ifvg_ts_3m is not None:
            ifvg_ts, ifvg_hi, ifvg_lo = ifvg_ts_3m, ifvg_hi_3m, ifvg_lo_3m

        # Find CISD
        cisd_ts, cisd_level = find_cisd(nq_5m_arrs, smt_ts, direction, None, 'CISD')

        # Both must be present
        if ifvg_ts is None or cisd_ts is None:
            continue

        # Entry after whichever completes second
        last_condition_ts = max(ifvg_ts, cisd_ts)

        # Entry = next 5m candle open after last condition
        entry_5m_idx = int(np.searchsorted(nq_5m_arrs['ts_ns'], last_condition_ts, side='right'))
        if entry_5m_idx >= len(nq_5m_arrs['ts_ns']):
            continue

        entry_ts_ns = int(nq_5m_arrs['ts_ns'][entry_5m_idx])
        entry_price = float(nq_5m_arrs['open'][entry_5m_idx])

        # Check entry is still in session
        entry_1m_idx = int(np.searchsorted(m1_ts, entry_ts_ns, side='left'))
        if entry_1m_idx >= len(m1_ts):
            continue
        entry_hr = int(m1_hr[entry_1m_idx])
        entry_mn = int(m1_mn[entry_1m_idx])
        entry_hr_float = entry_hr + entry_mn / 60.0
        if entry_hr_float < SESSION_HRS[0] or entry_hr_float >= SESSION_HRS[1]:
            continue

        # Stop = IFVG invalidation level
        if direction == 'LONG':
            stop_price = ifvg_lo  # close below IFVG low invalidates
            if entry_price <= stop_price:

            'mfe_pct': None,

            'target_price': target_price,
```

### Task 8: Outcome Resolution (IFVG Close-Based Stop)

**Files:**
- Modify: `555/model_stats.py`

- [ ] **Step 1: Write outcome resolution with close-based SL**

This is the key difference from Fractal Sweep — SL is checked on 5m **closes**, not intra-candle highs/lows. TP1 and runner BE stop still use highs/lows.

```python
def resolve_outcomes(nq_5m_arrs, pending):
    """
    Resolve trade outcomes for the 555 model.

    Key difference from Fractal Sweep:
      - SL is checked on 5m candle CLOSES only (IFVG invalidation)
      - TP1 is checked on highs/lows (standard)
      - Runner BE stop is checked on highs/lows

    Returns list of (outcome, net_r, mae_pct, mfe_pct, tp1_hit, runner_exit_r)
    """
    ts_ns = nq_5m_arrs['ts_ns']
    highs = nq_5m_arrs['high']
    lows = nq_5m_arrs['low']
    closes = nq_5m_arrs['close']
    N = len(ts_ns)

    results = []
    for e in pending:
        entry_ts_ns = e['entry_ts_ns']
        entry_price = e['entry_price']
        stop_price = e['stop_price']
        target_price = e['target_price']
        direction = e['direction']
        ifvg_high = e['ifvg_high']
        ifvg_low = e['ifvg_low']

        risk = abs(entry_price - stop_price)
        if risk > MAX_RISK_PTS:
            results.append(('INVALID', 0.0, 0.0, 0.0, False, 0.0))
            continue

        start = int(np.searchsorted(ts_ns, entry_ts_ns, side='right'))
        end = min(start + OUTCOME_MAX_BARS, N)

        if start >= N:
            results.append(('EXPIRED', 0.0, 0.0, 0.0, False, 0.0))
            continue

        h = highs[start:end]
        l = lows[start:end]
        c = closes[start:end]

        # MAE/MFE over full window
        if direction == 'LONG':
            mae_pct = round(float(max(0.0, entry_price - l.min()) / entry_price * 100), 4)
            mfe_pct = round(float(max(0.0, h.max() - entry_price) / entry_price * 100), 4)
        else:
            mae_pct = round(float(max(0.0, h.max() - entry_price) / entry_price * 100), 4)
            mfe_pct = round(float(max(0.0, entry_price - l.min()) / entry_price * 100), 4)

        # Bar-by-bar resolution: SL on closes, TP1 on highs/lows
        tp1_idx = None
        sl_idx = None
        for bar in range(len(h)):
            # Check TP1 (intra-candle)
            if tp1_idx is None:
                if direction == 'LONG' and h[bar] >= target_price:
                    tp1_idx = bar
                elif direction == 'SHORT' and l[bar] <= target_price:
                    tp1_idx = bar
            # Check SL (on close only — IFVG invalidation)
            if sl_idx is None:
                if direction == 'LONG' and c[bar] < ifvg_low:
                    sl_idx = bar
                elif direction == 'SHORT' and c[bar] > ifvg_high:
                    sl_idx = bar
            # If both found, stop scanning
            if tp1_idx is not None and sl_idx is not None:
                break

        # Determine outcome
        if tp1_idx is None and sl_idx is None:
            # Expired
            last_r = ((c[-1] - entry_price) / risk if direction == 'LONG'
                      else (entry_price - c[-1]) / risk)
            results.append(('EXPIRED', round(float(last_r), 2), mae_pct, mfe_pct, False, 0.0))
            continue

        if sl_idx is not None and (tp1_idx is None or sl_idx < tp1_idx):
            # SL hit before TP1 — compute actual R at stop
            # Since SL is close-based, the actual loss may differ from -1.0
            if direction == 'LONG':
                actual_r = (c[sl_idx] - entry_price) / risk
            else:
                actual_r = (entry_price - c[sl_idx]) / risk
            results.append(('LOSS', round(float(actual_r), 2), mae_pct, mfe_pct, False, 0.0))
            continue

        # Same bar: TP1 and SL on same bar — assume TP1 hit first (conservative: intra-candle)
        if tp1_idx is not None and sl_idx is not None and tp1_idx == sl_idx:
            # TP1 checked on wick, SL on close — TP1 likely hit first within the bar
            pass  # fall through to TP1 logic

        # TP1 hit — run the runner with BE stop
        runner_start = tp1_idx + 1
        runner_end = len(h)
        runner_exit_r = 0.0

        if runner_start < runner_end:
            rh = h[runner_start:runner_end]
            rl = l[runner_start:runner_end]
            if direction == 'LONG':
                be_hit = rl <= entry_price
            else:
                be_hit = rh >= entry_price

            if not be_hit.any():
                last_close = c[runner_start + len(rh) - 1]
                if direction == 'LONG':
                    runner_exit_r = (last_close - entry_price) / risk
                else:
                    runner_exit_r = (entry_price - last_close) / risk
                runner_exit_r = max(0.0, round(float(runner_exit_r), 3))

        net_r = round(0.90 * 1.0 + 0.10 * runner_exit_r, 3)
        results.append(('WIN', net_r, mae_pct, mfe_pct, True, runner_exit_r))

    return results
```

```python
def main():
```
